\chapter{Misura}
\section{Misura singola}
\begin{itemize}
	\item $\overline{x}$ Misura
	\item $\Delta x$ incertezza strumento
\end{itemize}\index{Misura!singola}
\[x=\overline{x}\pm\Delta x\]
\section{Misura multipla}
\begin{itemize}
	\item $\overline{x}$ Media misure
	\item $\Delta x=\dfrac{x_{max}-x_{min}}{e}$ Errore assoluto
\end{itemize}\index{Misura!multipla}
\[x=\overline{x}\pm\Delta x\]
\section{Incertezza relativa}
\[e_r=\dfrac{\Delta x}{\overline{x}}\]\index{Incertezza!relativa}
\section{Incertezza Percentuale}\index{Incertezza!percentuale}
\[e_p=e_r\cdot 100\%\]
