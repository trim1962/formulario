\chapter{La percentuale}
\section{Vocabolario}
\begin{description}
	\item[r] ragione o tasso
	\item[p] percentuale
	\item[T] totale
\end{description}
\begin{align}
r:100=&p:T\\
p=&\dfrac{T\cdot r}{100}\\
r=&\dfrac{p\cdot100}{T}\\
T=&\dfrac{p\cdot100}{r}\\
\end{align}\index{Percentuale}
